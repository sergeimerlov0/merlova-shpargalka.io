простой таб

window.addEventListener('DOMContentLoaded', () => {
// прописываем дом чтобы страница не перезапускалась
   const contans = document.querySelectorAll('. block__text'),
      tabs = document.querySelectorAll('.block__tab'),
      tabPerent = document.querySelector('.block__tabs');
// задаем переменные каждого таба, родителя табов и контента
function removeTabContent() {
      contans.forEach(item => {
            item.style.display = 'none';
});
tabs.forEach(item => {
      item.classList.remove('block__tab_activ');
});
}

function showTabContent(i = 0) {
      contans[i].style.display = 'block';
      tabs[i].classList.add('block__tab_activ');
}
// прописываем две функции, первая отменяет весь контент и актывные табы, вторая показывет первый контент и первый таб из блока
removeTabContent();
showTabContent();
// вызываем обе функции

tabPerent.addEventListener('click', (event) => {
// вызываем обработчик события
   const target = event.target;
// создаем переменную target , чтобы каждый раз не прописывать полностью event.target
if(target && target.classList.contains('block__tab')) {
// проверяем действительно ли мы кликнули в таб
      tabs.forEach((item, i) => {
// мы перебераем все табы, item - каждый таб, i - номер таба
            if(target == item) {
// если target - который мы кликнули , будет совподать с элементом то --
      removeTabContent();
      showTabContent(i);
// -- мы будем вызывать эти две функции
}
});
}
});
});
