class (ES6)

class constructorDiv {

//забается имя классу
   constructor(subtitle, text, parentSelector) {
//создается конструктор
   this.subtitle = subtitle;
   this.text = text;
   this.parent = document.querySelector
(parentSelector);
}
render() {
//делается верстка
   const element = document.createElement('div');
   element.innerHTML = `
div class="box1_title">
   div class="box1_subtitle">${this.subtitle}/div>
   div class="box1_text">${this.text}/div> /div>
`;
   this.parent.append(element);
}
}

new constructorDiv(
//задаются параметры аргументам

   ".box1"
).render();
